#include <stdio.h>
#include<malloc.h>
#include<ctype.h>

struct stack
{
  int data;
  struct stack* next;
};

struct stack *top,*top1,*top2,*top3,*start;

struct stack* push(struct stack* top,int val)
{
  struct stack* newnode;
  newnode=(struct stack*)malloc(sizeof(struct stack));
  newnode->data=val;
  if(top==NULL)
  {
    top=newnode;
    newnode->next=NULL;
  }
  else
  {
     newnode->next=top;
     top=newnode;
  }
  return top;
}

int display(struct stack* top)
{
  struct stack* ptr=top;
  if(top==NULL)
  return 0;
  else{
  while(ptr!=NULL)
  {
    if(isdigit(ptr->data))
    printf("%d ",ptr->data-48);
    else
    printf("%c ",ptr->data);
    ptr=ptr->next;
  }
 }
  return 1;
}
